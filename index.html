            const generateReport = async (results) => {

                const validResults = results.filter(r => !r.error);

                

                if (validResults.length === 0) {

                    return {

                        passed: false,

                        overallScore: 0,

                        message: "Testing failed - no valid results"

                    };

                }


                const scores = validResults.map(r => r.evaluation.overallScore);

                const avgScore = scores.reduce((a, b) => a + b, 0) / scores.length;


                const brandVoiceScores = validResults.map(r => r.evaluation.brandVoice);

                const knowledgeScores = validResults.map(r => r.evaluation.knowledgeAccuracy);

                const goalScores = validResults.map(r => r.evaluation.goalAlignment);


                const avgBrandVoice = brandVoiceScores.reduce((a, b) => a + b, 0) / brandVoiceScores.length;

                const avgKnowledge = knowledgeScores.reduce((a, b) => a + b, 0) / knowledgeScores.length;

                const avgGoals = goalScores.reduce((a, b) => a + b, 0) / goalScores.length;


                const allWeaknesses = validResults.flatMap(r => r.evaluation.weaknesses);

                const allCriticalIssues = validResults.flatMap(r => r.evaluation.criticalIssues);


                const reportPrompt = `Generate a final testing report based on these results:


OVERALL SCORE: ${avgScore.toFixed(1)}%

TESTS RUN: ${validResults.length}

PASS THRESHOLD: 90%


COMPONENT SCORES:

- Brand Voice: ${avgBrandVoice.toFixed(1)}%

- Knowledge Accuracy: ${avgKnowledge.toFixed(1)}%

- Goal Alignment: ${avgGoals.toFixed(1)}%


WEAKNESSES IDENTIFIED:

${allWeaknesses.join('\n')}


CRITICAL ISSUES:

${allCriticalIssues.length > 0 ? allCriticalIssues.join('\n') : 'None'}


Provide a summary and specific recommendations for improving the Brand Voice, Knowledge Base, and Bot Goals documents. Be specific and actionable.

                    method: "POST",

                        "Content-Type": "application/json",

                    },

                        model: "claude-sonnet-4-20250514",

                        max_tokens: 1000,

                        messages: messages,

                    })

                });


                if (!response.ok) {

                    throw new Error(`API call failed: ${response.statusText}`);

                }


                return await response.json();

            };
